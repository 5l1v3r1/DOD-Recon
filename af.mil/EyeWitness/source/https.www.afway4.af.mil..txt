        function sadTime() {

            var depth = [],
                identifiers = {},
                hashMap = {

                },
                // private scope
                __ = {
                    extend: function () {
                        var parts = depth,
                            parent = hashMap;

                        for (var i = 0; i < parts.length; i++) {
                            //create a property if it doesnt exist
                            if (typeof parent[parts[i]] == 'undefined') {
                                parent[parts[i]] = {};
                            }

                            parent = parent[parts[i]];
                        }

                        return parent;
                    },
                    inner: function () {
                        var parts = depth,
                            parent = hashMap;

                        for (var i = 0; i < parts.length; i++) {
                            parent = parent[parts[i]];
                        }

                        return parent;
                    },
                    addOne: function (error, prop) {
                        prop = depth.length > 0 ? depth.join('.') + '.' + prop : prop;
                        hashMap[prop] = error;
                    },
                    addMany: function (error, props) {
                        props.forEach(function (prop) {
                            __.extend()[prop] = error;
                        });
                    },
                    forOne: function (name, optID) {
                        var args = Array.prototype.slice.call(arguments);
                        depth.push(name);
                        if (optID) {
                            identifiers[name] = optID;
                        }
                    },
                };

            this.add = function (error, props) {
                var args = Array.prototype.slice.call(arguments);
                if (Array.isArray(props)) {
                    // .add('oh no', ['rfqID','contractNo']);
                    __.addMany(error, props);
                } else if (typeof props == 'string') {
                    // .add('oh no', 'contractNo');
                    __.addOne(error, props);
                }

                return this;
            };
            this.for = function (name, optID) {
                __.forOne(name, optID);
                return this;
            };
            this.parent = function (error, props) {
                var removed = depth.pop();
                // remove them if they exist.
                delete identifiers[removed];

                return this.add(error, props);
            };
            this.revert = function () {
                var removed = depth.pop();
                // remove them if they exist.
                delete identifiers[removed];

                return this;
            };
            this.toString = function () {
                return JSON.stringify(hashMap);
            };

            this.errors = hashMap;
        };
